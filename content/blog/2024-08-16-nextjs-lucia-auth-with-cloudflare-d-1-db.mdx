  const user = await db
    .insert(usersTable)

  if (!user || user.length === 0) {
    throw new Error("An error occured during sign up.");
  }

  const lucia = initializeLucia(DB);
  const session = await lucia.createSession(user[0].id, {});

        sessionCookie.value,
